# 每日市场扫描生成系统

## 系统概述

基于AKShare/TuShare数据源和现有datasource框架的每日自动化市场扫描报告生成系统。

**系统目标**：
- 每日自动生成标准化市场扫描报告
- 整合多数据源，提供可追溯的技术分析
- 支持Pring六阶段周期判断
- 提供完整的资金流向分析
- **新增**: 库存周期验证机制，确保商品趋势判断的基本面支撑

## 技术指标计算体系

### 1. 移动平均线系统
```python
def calculate_moving_averages(df):
    """计算移动平均线"""
    df['MA20'] = df['close'].rolling(window=20).mean()
    df['MA50'] = df['close'].rolling(window=50).mean()
    df['MA200'] = df['close'].rolling(window=200).mean()
    return df

def calculate_ma_slope(df, period=5):
    """计算均线斜率"""
    ma20_slope = (df['MA20'].iloc[-1] - df['MA20'].iloc[-period-1]) / period
    ma50_slope = (df['MA50'].iloc[-1] - df['MA50'].iloc[-period-1]) / period
    return ma20_slope, ma50_slope
```

### 2. 波动率计算
```python
def calculate_volatility(df, window=30):
    """计算年化波动率"""
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=window).std() * np.sqrt(252) * 100
    return volatility.iloc[-1]
```

### 3. 趋势评分系统
```python
def calculate_trend_score(price, ret_30d, ma50, ma200, ma20_slope):
    """Pring趋势评分系统 (-2至+2)"""
    score = 0
    
    # 近30日收益 >= 1% (+1分)
    if ret_30d >= 1.0:
        score += 1
    elif ret_30d <= -1.0:
        score -= 1
        
    # 收盘高于MA50 (+1分)
    if price > ma50:
        score += 1
    elif price < ma50:
        score -= 1
        
    # MA50 > MA200 (+1分)
    if ma50 > ma200:
        score += 1
    elif ma50 < ma200:
        score -= 1
        
    # MA20斜率为正 (+1分)
    if ma20_slope > 0:
        score += 1
    elif ma20_slope < 0:
        score -= 1
    
    return max(-2, min(2, score))
```

### 4. 库存周期验证系统 (新增)
```python
def verify_commodity_bullish(price_data, ppi_data, pmi_data, bdi_data):
    """商品Bullish双重验证 (技术面35% + 库存周期65%)"""
    
    # 技术面评分 (35%权重)
    technical_score = calculate_technical_score(price_data)
    
    # 库存周期评分 (60%权重)
    inventory_score = 0
    
    # PPI环比连续为正验证 (25分)
    ppi_info = analyze_ppi_trend(ppi_data)
    if ppi_info['consecutive_months'] >= 2:
        inventory_score += 25
    
    # PPI同比改善验证 (15分)
    if ppi_info['latest_yoy'] > -0.5:  # 降幅收窄
        inventory_score += 15
    
    # PMI库存去化验证 (20分)
    pmi_inventory = analyze_pmi_inventory(pmi_data)
    if pmi_inventory and pmi_inventory['stage'] == "去库存阶段":
        inventory_score += 20
    
    # 综合评分
    total_score = technical_score * 0.35 + inventory_score * 0.65
    
    if total_score >= 70:
        verdict = "Bullish"
    elif total_score <= 30:
        verdict = "Bearish" 
    else:
        verdict = "N/A(验证不充分)"
    
    return {
        'technical_score': technical_score,
        'inventory_score': inventory_score, 
        'total_score': total_score,
        'verdict': verdict
    }

def analyze_inventory_cycle_stage(ppi_yoy, pmi_inventory):
    """库存周期阶段判断"""
    if ppi_yoy > 0 and pmi_inventory < 50:
        return "主动补库存", "强牛"
    elif ppi_yoy > 0 and pmi_inventory > 50:
        return "被动补库存", "偏牛"
    elif ppi_yoy < 0 and pmi_inventory < 50:
        return "主动去库存", "中性"
    elif ppi_yoy < 0 and pmi_inventory > 50:
        return "被动去库存", "熊"
    else:
        return "待确认", "N/A"
```

## 每日生成流程

### 主流程 (market_scan_generator.py)
```python
async def generate_daily_report():
    """每日报告生成主流程"""
    
    # 1. 初始化系统
    manager = await initialize_default_manager()
    config = load_config()
    
    # 2. 数据收集阶段
    print("开始数据收集...")
    a_stock_data = await collect_a_stock_data(manager, config)    # A股指数
    hk_stock_data = await collect_hk_stock_data(manager, config)  # 港股指数
    us_stock_data = await collect_us_stock_data(manager, config)  # 美股指数
    commodity_data = await collect_commodity_data(manager, config)  # 大宗商品
    forex_data = await collect_forex_data(manager, config)      # 汇率
    bond_data = await collect_bond_data(manager, config)        # 债券利率
    fund_flow_data = await collect_fund_flow_data(manager, config)  # 资金流向
    
    # 3. 技术分析阶段
    print("计算技术指标...")
    all_stock_data = {**a_stock_data, **hk_stock_data, **us_stock_data}
    technical_results = calculate_technical_indicators(all_stock_data)
    trend_scores = calculate_trend_scores(technical_results)
    volatility_analysis = calculate_volatility_analysis(all_stock_data)
    
    # 4. Pring分析阶段
    print("进行Pring六阶段分析...")
    pring_analysis = analyze_pring_cycle(
        bonds=bond_data,
        stocks=stock_data,
        commodities=commodity_data
    )
    
    # 5. 库存周期验证阶段 (新增)
    print("进行库存周期验证...")
    macro_data = await collect_macro_data(manager, config)  # PPI、CPI、PMI
    inventory_cycle_analysis = verify_commodity_bullish(
        commodity_data,
        macro_data['ppi'],
        macro_data['pmi'],
        macro_data['bdi']
    )
    
    # 6. 报告生成阶段
    print("生成报告...")
    report_data = {
        'stocks': technical_results,
        'commodities': commodity_data,
        'forex': forex_data,
        'bonds': bond_data,
        'fund_flows': fund_flow_data,
        'pring': pring_analysis,
        'inventory_cycle': inventory_cycle_analysis,  # 新增
        'macro_data': macro_data,  # 新增
        'metadata': {
            'generation_time': datetime.now().strftime("%Y-%m-%d %H:%M"),
            'data_range': get_data_range(),
            'inventory_verification': True  # 标记包含库存周期验证
        }
    }
    
    # 7. 格式化和输出
    markdown_report = generate_markdown_report(report_data, config)
    
    # 8. 保存报告
    filename = f"{datetime.now().strftime('%Y%m%d')}-市场扫描.md"
    save_report(markdown_report, filename)
    
    print(f"报告已生成: {filename}")
    return filename
```

async def collect_a_stock_data(manager, config):
    """收集A股指数数据"""
    results = {}
    
    for category, assets in config['assets']['A股指数'].items():
        try:
            # 获取历史数据（用于技术指标计算）
            end_date = datetime.now().strftime("%Y-%m-%d")
            start_date = (datetime.now() - timedelta(days=300)).strftime("%Y-%m-%d")
            
            data = await manager.get_index_daily(
                assets['akshare_code'], 
                start_date, 
                end_date
            )
            
            if not data.error:
                results[category] = {
                    'data': data.data,
                    'source': data.source,
                    'display_name': assets['display_name']
                }
                
        except Exception as e:
            print(f"获取{category}数据失败: {e}")
            results[category] = {'error': str(e)}
    
    return results
```

                'data': pmi_data.data,
                'source': pmi_data.source,
                'latest_pmi': pmi_data.data.iloc[-1]['制造业PMI'] if not pmi_data.data.empty else None,
                'raw_materials_inventory': pmi_data.data.iloc[-1].get('原材料库存', None),
                'finished_goods_inventory': pmi_data.data.iloc[-1].get('产成品库存', None)
            }
        
        # BDI指数
        bdi_data = await manager.get_index_data('bdi')
        if not bdi_data.error:
            results['bdi'] = {
                'data': bdi_data.data,
                'source': bdi_data.source,
                'latest_value': bdi_data.data.iloc[-1]['close'] if not bdi_data.data.empty else None,
                'change_30d': calculate_percentage_change(bdi_data.data, 30)
            }
            
    except Exception as e:
        print(f"获取宏观数据失败: {e}")
        results['error'] = str(e)
    
    return results
```

## 库存周期验证框架

### 验证标准
- **技术面评分**: 价格趋势、成交量确认、BDI指数变化
- **库存周期评分**: PPI环比连续性、PMI库存分项、需求指标验证
- **综合判定**: ≥70分Bullish、≤30分Bearish、中间N/A
- **配置调整**: 根据验证结果调整商品权重配置

### 实战应用
2025年9月10日验证案例显示，技术面商品价格上涨，但库存周期验证显示PPI深度通缩、PMI收缩，综合评分仅27分判定为Bearish，成功避免了基于技术面误判的投资风险。
