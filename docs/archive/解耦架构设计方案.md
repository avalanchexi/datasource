# 解耦架构设计方案 - Pring分析独立化

**版本**: V3.1 (解耦版)
**创建日期**: 2025-11-10
**核心理念**: 单一职责原则 - 数据收集与分析分离

---

## 一、解耦设计理念

### 1.1 为什么要解耦？

**当前问题**（耦合架构）:
```
background_scan_120d_generator.py (1577行)
    ├── 数据收集
    ├── Pring分析         <-- 耦合在一起
    ├── 报告生成
    └── 质量验证
```

**问题**:
- ❌ 职责不清：一个脚本做太多事
- ❌ 难以测试：无法单独测试Pring分析
- ❌ 难以复用：Pring分析无法用于其他场景
- ❌ 难以维护：修改Pring逻辑可能影响数据收集
- ❌ 难以扩展：无法灵活组合不同的分析模块

**解耦后**（管道架构）:
```
数据收集脚本 → 数据JSON文件 → Pring分析脚本 → 分析结果JSON → 报告生成脚本
     ↓              ↓               ↓                ↓              ↓
数据采集器      标准数据格式      独立分析器        分析结果       报告生成器
(采集职责)      (接口契约)       (分析职责)       (结果契约)     (呈现职责)
```

**优势**:
- ✅ 职责清晰：每个脚本只做一件事
- ✅ 易于测试：可以用测试数据验证Pring逻辑
- ✅ 高度复用：Pring分析可用于实时监控、回测等
- ✅ 易于维护：修改互不影响
- ✅ 灵活扩展：可以添加其他分析模块（技术面、基本面等）

## 四、独立脚本设计

### 4.1 pring_analyzer_standalone.py

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
Pring分析器 - 独立脚本
完全解耦，可独立运行，可复用
"""
import asyncio
import json
import sys
from pathlib import Path
from datetime import datetime
from typing import Dict, Any, Optional

# 最小化依赖 - 仅依赖Pring分析器核心
from datasource.calculators.pring_analyzer import PringAnalyzer
from datasource import get_manager


class StandalonePringAnalyzer:
    """独立的Pring分析器 - 从JSON读取数据"""

    def __init__(self, data_file: str):
        """
        初始化分析器

        Args:
            data_file: market_data.json文件路径
        """
        self.data_file = Path(data_file)
        self.market_data = None
        self.pring_analyzer = PringAnalyzer(get_manager())

    def load_market_data(self) -> Dict[str, Any]:
        """加载市场数据JSON"""
        print(f"📂 加载市场数据: {self.data_file}")

        if not self.data_file.exists():
            raise FileNotFoundError(f"数据文件不存在: {self.data_file}")

        with open(self.data_file, 'r', encoding='utf-8') as f:
            self.market_data = json.load(f)

        print(f"✅ 数据加载成功，包含 {len(self.market_data)} 个数据类别")
        return self.market_data

    def validate_data_completeness(self) -> bool:
        """验证Pring分析所需数据是否完整"""
        print("\n🔍 验证Pring分析数据完整性...")

        required_data = {
            'macro_indicators.inventory_cycle': ['PPI', 'PMI', 'industrial_production', 'BDI', 'CPI'],
            'macro_indicators.monetary_cycle': ['reserve_ratio', 'reverse_repo_rate', 'TSF_growth', 'M2_growth'],
            'stock_indices': ['000300', '399006'],  # 沪深300、创业板指
            'commodities': ['COMEX_GOLD', 'COMEX_COPPER', 'BCOM'],
            'bonds': ['CN10Y', 'US10Y']
        }

        missing = []
        for path, fields in required_data.items():
            parts = path.split('.')
            data = self.market_data

            # 导航到嵌套字段
            for part in parts:
                if part not in data:
                    missing.append(f"{path} (缺失)")
                    data = {}
                    break
                data = data[part]

            # 检查必需字段
            if data:
                for field in fields:
                    if field not in data:
                        missing.append(f"{path}.{field}")

        if missing:
            print(f"❌ 缺失必需数据: {', '.join(missing)}")
            return False

        print("✅ 所有必需数据完整")
        return True

    async def analyze(self) -> Dict[str, Any]:
        """执行Pring V4.0三层框架分析"""
        print("\n🔬 开始Pring V4.0三层框架分析...")

        # 从market_data提取Pring输入数据
        pring_input = self._prepare_pring_input()

        # 执行三层分析
        result = await self._execute_three_layer_analysis(pring_input)

        return result

    def _prepare_pring_input(self) -> Dict[str, Any]:
        """从market_data准备Pring分析输入"""
        print("  准备Pring输入数据...")

        macro = self.market_data['macro_indicators']

        return {
            'inventory_cycle': macro['inventory_cycle'],
            'monetary_cycle': macro['monetary_cycle'],
            'stock_signal': self._calculate_stock_signal(),
            'commodity_signal': self._calculate_commodity_signal(),
            'bond_signal': self._calculate_bond_signal()
        }

    def _calculate_stock_signal(self) -> str:
        """计算股票信号"""
        stocks = self.market_data['stock_indices']
        hs300 = stocks.get('000300', {})
        cyb = stocks.get('399006', {})

        hs300_chg = hs300.get('change_120d_pct', 0)
        cyb_chg = cyb.get('change_120d_pct', 0)

        avg_change = (hs300_chg + cyb_chg) / 2

        if avg_change > 10:
            return "Bullish"
        elif avg_change < -10:
            return "Bearish"
        else:
            return "Neutral"

    def _calculate_commodity_signal(self) -> str:
        """计算商品信号"""
        commodities = self.market_data['commodities']

        gold = commodities.get('COMEX_GOLD', {}).get('ytd_change_pct', 0)
        copper = commodities.get('COMEX_COPPER', {}).get('ytd_change_pct', 0)
        bcom = commodities.get('BCOM', {}).get('ytd_change_pct', 0)

        # 综合判断
        bullish_count = sum([gold > 20, copper > 10, bcom > 5])

        if bullish_count >= 2:
            return "Bullish"
        elif bullish_count == 1:
            return "Mixed"
        else:
            return "Bearish"

    def _calculate_bond_signal(self) -> str:
        """计算债券信号"""
        bonds = self.market_data['bonds']

        cn10y_chg = bonds.get('CN10Y', {}).get('change_120d_bp', 0)
        us10y_chg = bonds.get('US10Y', {}).get('change_120d_bp', 0)

        # 收益率下降 = 债券价格上涨 = Bullish
        if cn10y_chg < -10 and us10y_chg < -20:
            return "Bullish"
        elif cn10y_chg > 10 and us10y_chg > 20:
            return "Bearish"
        else:
            return "Neutral"

    async def _execute_three_layer_analysis(self, pring_input: Dict) -> Dict[str, Any]:
        """执行三层分析"""

        # 第一层：库存周期分析
        print("  [第一层] 库存周期分析...")
        layer_1 = self._analyze_inventory_cycle(pring_input['inventory_cycle'])

        # 第二层：货币周期叠加
        print("  [第二层] 货币周期叠加...")
        layer_2 = self._analyze_monetary_cycle(pring_input['monetary_cycle'])

        # 第三层：Pring六阶段判定
        print("  [第三层] Pring六阶段判定...")
        layer_3 = self._determine_pring_stage(
            pring_input['bond_signal'],
            pring_input['stock_signal'],
            pring_input['commodity_signal'],
            layer_1,
            layer_2
        )

        # 资产配置建议
        allocation = self._generate_asset_allocation(layer_3)

        return {
            'metadata': {
                'analysis_date': datetime.now().strftime('%Y-%m-%d'),
                'framework_version': 'V4.0',
                'analysis_time': datetime.now().isoformat()
            },
            'layer_1_inventory_cycle': layer_1,
            'layer_2_monetary_cycle': layer_2,
            'layer_3_pring_final': layer_3,
            'asset_allocation': allocation,
            'confidence_breakdown': {
                'base_confidence': layer_3.get('base_confidence', 0.70),
                'monetary_cycle_boost': layer_2.get('confidence_boost', 0.05),
                'final_confidence': layer_3.get('final_confidence', 0.75)
            }
        }

    # ... 其他分析方法实现 ...

    def save_result(self, result: Dict[str, Any], output_file: str):
        """保存分析结果"""
        output_path = Path(output_file)
        output_path.parent.mkdir(parents=True, exist_ok=True)

        with open(output_path, 'w', encoding='utf-8') as f:
            json.dump(result, f, ensure_ascii=False, indent=2)

        print(f"\n✅ 分析结果已保存: {output_path}")


async def main():
    """主函数"""
    import argparse

    parser = argparse.ArgumentParser(description='Pring V4.0独立分析器')
    parser.add_argument('--input', required=True, help='market_data.json路径')
    parser.add_argument('--output', required=True, help='pring_result.json输出路径')
    parser.add_argument('--verbose', action='store_true', help='详细输出')

    args = parser.parse_args()

    try:
        # 初始化分析器
        analyzer = StandalonePringAnalyzer(args.input)

        # 加载数据
        analyzer.load_market_data()

        # 验证数据完整性
        if not analyzer.validate_data_completeness():
            print("\n❌ 数据不完整，无法进行Pring分析")
            return 1

        # 执行分析
        result = await analyzer.analyze()

        # 保存结果
        analyzer.save_result(result, args.output)

        # 打印摘要
        print("\n" + "="*60)
        print("📊 Pring分析摘要")
        print("="*60)
        print(f"最终阶段: {result['layer_3_pring_final']['final_stage']}")
        print(f"置信度: {result['layer_3_pring_final']['final_confidence']:.0%}")
        print(f"解释: {result['layer_3_pring_final']['interpretation']}")
        print("="*60)

        return 0

    except Exception as e:
        print(f"\n❌ 分析失败: {e}")
        import traceback
        traceback.print_exc()
        return 1


if __name__ == '__main__':
    sys.exit(asyncio.run(main()))
```
